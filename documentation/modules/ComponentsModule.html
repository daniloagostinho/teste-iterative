import { ShowComicsComponent } from './show-comics/show-comics.component';

import { RatingComponent } from './rating/rating.component';
import { ConfirmPagamentoComponent } from './confirm-pagamento/confirm-pagamento.component';
import { NumberPikerComponent } from './number-piker/number-piker.component';

@NgModule({
  imports: [
    CommonModule
  ],
  declarations: [
    BarHomeComponent,
    QuantidadeComicsComponent,
    DescriptionComponent,
    FeedbackUsuarioComponent,
    HomeComponent,
    ListComicsComponent,
    ShowComicsComponent,

    BarHomeComponent,

    HomeComponent,
    ListComicsComponent,
    ShowComicsComponent,
    RatingComponent,
    ConfirmPagamentoComponent,
    NumberPikerComponent],
  schemas: [ CUSTOM_ELEMENTS_SCHEMA ]
})
export class ComponentsModule { }
